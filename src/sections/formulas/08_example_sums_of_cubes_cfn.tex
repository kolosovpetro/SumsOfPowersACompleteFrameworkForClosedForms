\begin{example} [Sums of cubes in central factorial numbers]
  \begin{align*}
    t=0:\quad
    \KnuthRFoldSum{1}{n}{4}
    &= n\cdot0 + \tfrac12 n(1+n)\cdot0 + \tfrac16 n(1+3n+2n^2)\cdot1
    \\
    &\quad + \tfrac18 n(-1+n+4n^2+2n^3)\cdot0
    + \tfrac1{10} n(-2-5n+5n^3+2n^4)\cdot1,
    \\[4pt]
    %
    t=1:\quad
    \KnuthRFoldSum{1}{n}{4}
    &= n\cdot1 + \tfrac12 n(n-1)\cdot5 + \tfrac16 n(1-3n+2n^2)\cdot7
    \\
    &\quad + \tfrac18 n(1+n-4n^2+2n^3)\cdot4
    + \tfrac1{10} n(-2+5n-5n^3+2n^4)\cdot1,
    \\[4pt]
    %
    t=2:\quad
    \KnuthRFoldSum{1}{n}{4}
    &= n\cdot16 + \tfrac12 n(n-3)\cdot34 + \tfrac16 n(13-9n+2n^2)\cdot25
    \\
    &\quad + \tfrac18 n(-21+25n-12n^2+2n^3)\cdot8
    + \tfrac1{10} n(18-45n+40n^2-15n^3+2n^4)\cdot1,
    \\[4pt]
    %
    t=3:\quad
    \KnuthRFoldSum{1}{n}{4}
    &= n\cdot81 + \tfrac12 n(n-5)\cdot111 + \tfrac16 n(37-15n+2n^2)\cdot55
    \\
    &\quad + \tfrac18 n(-115+73n-20n^2+2n^3)\cdot12
    + \tfrac1{10} n(298-275n+120n^2-25n^3+2n^4)\cdot1,
    \\[4pt]
    %
    t=4:\quad
    \KnuthRFoldSum{1}{n}{4}
    &= n\cdot256 + \tfrac12 n(n-7)\cdot260 + \tfrac16 n(73-21n+2n^2)\cdot97
    \\
    &\quad + \tfrac18 n(-329+145n-28n^2+2n^3)\cdot16
    + \tfrac1{10} n(1318-805n+240n^2-35n^3+2n^4)\cdot1,
    \\[4pt]
    %
    t=5:\quad
    \KnuthRFoldSum{1}{n}{4}
    &= n\cdot625 + \tfrac12 n(n-9)\cdot505 + \tfrac16 n(121-27n+2n^2)\cdot151
    \\
    &\quad + \tfrac18 n(-711+241n-36n^2+2n^3)\cdot20
    + \tfrac1{10} n(3798-1755n+400n^2-45n^3+2n^4)\cdot1.
  \end{align*}
\end{example}
