\begin{example} [Sums of squares]
  \begin{align*}
    t=0:\quad
    \KnuthRFoldSum{1}{n}{2}
    &=
    \frac{t^2}{2}(2n)
    +
    \frac{\delta t^2}{2}\, n(1+n)
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(1+3n+2n^2)}{6},
    \\[8pt]
    %
    t=1:\quad
    \KnuthRFoldSum{1}{n}{2}
    &=
    \frac{t^2}{2}(2n)
    +
    \frac{\delta t^2}{2}\, n(n-1)
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(1-3n+2n^2)}{6},
    \\[8pt]
    %
    t=2:\quad
    \KnuthRFoldSum{1}{n}{2}
    &=
    \frac{t^2}{2}(2n)
    +
    \frac{\delta t^2}{2}\, n(n-3)
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(13-9n+2n^2)}{6},
    \\[8pt]
    %
    t=3:\quad
    \KnuthRFoldSum{1}{n}{2}
    &=
    \frac{t^2}{2}(2n)
    +
    \frac{\delta t^2}{2}\, n(n-5)
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(37-15n+2n^2)}{6},
    \\[8pt]
    %
    t=4:\quad
    \KnuthRFoldSum{1}{n}{2}
    &=
    \frac{t^2}{2}(2n)
    +
    \frac{\delta t^2}{2}\, n(n-7)
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(73-21n+2n^2)}{6}.
  \end{align*}
\end{example}
