\begin{example}[Sums of cubes in central differences]
  \begin{align*}
    t=0:\quad
    \KnuthRFoldSum{1}{n}{3}
    &=
    \tfrac{t^3}{2}(2n)
    +
    \tfrac{\delta t^3}{2}\, n(1+n)
    +
    \tfrac{\delta^2 t^3}{2}\,\tfrac{n(1+3n+2n^2)}{6}
    +
    \tfrac{\delta^3 t^3}{2}\,\tfrac{n(-1+n+4n^2+2n^3)}{24},
    \\[8pt]

    t=1:\quad

    \\[8pt]
    %
    t=2:\quad
