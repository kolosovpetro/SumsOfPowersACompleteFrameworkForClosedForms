\begin{example}[Double sum of squares]
  \begin{align*}
    t=0:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \tfrac{t^2}{2}\, n(1+n)
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(2+3n+n^2)}{3}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(1+n)^2(2+n)}{12},
    \\[8pt]
    %
    t=1:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \tfrac{t^2}{2}\, n(1+n)
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-1+n^2)}{3}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n^2(-1+n^2)}{12},
    \\[8pt]
    %
    t=2:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \tfrac{t^2}{2}\, n(1+n)
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-4-3n+n^2)}{3}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(10+5n-4n^2+n^3)}{12},
    \\[8pt]
    %
    t=3:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \tfrac{t^2}{2}\, n(1+n)
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-7-6n+n^2)}{3}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(32+23n-8n^2+n^3)}{12},
    \\[8pt]
    %
    t=4:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \tfrac{t^2}{2}\, n(1+n)
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-10-9n+n^2)}{3}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(66+53n-12n^2+n^3)}{12}.
  \end{align*}
\end{example}
