In this manuscript, we derive formulas for sums of powers by combining
centered Newton's interpolation formula for powers
\eqref{lem:central-newtons-formula-for-powers}
with hockey-stick identities
\eqref{lem:generalized-hockey-stick-identity},
and
\eqref{lem:centered-hockey-stick-identity}.
This approach can be generalized further.
In particular, while using Newton's formulas
\begin{align}
  \label{eq:newtons-formulas-casess}
  \begin{cases}
    f(x)
    &= \sum_{k=0}^{\infty} \frac{\centralFactorial{(x-a)}{k}}{k!} \delta^{k} f(a)
    = \sum_{k=0}^{\infty} \frac{x-a}{k} \binom{x-a+\frac{k}{2}-1}{k-1} \delta^{k} f(a), \\
    f(x)
    &= \sum_{k=0}^{\infty} \frac{\fallingFactorial{x-a}{k}}{k!} \Delta^k f(a)
    = \sum_{k=0}^{\infty} \binom{x-a}{k} \Delta^k f(a), \\
    f(x)
    &= \sum_{k=0}^{\infty} \frac{\risingFactorial{(x-a)}{k}}{k!} \nabla^k f(a)
    = \sum_{k=0}^{\infty} \binom{x-a+k-1}{k} \nabla^k f(a), \\
    f(x)
    &= \sum_{k=0}^{\infty} \frac{(x-a)^k}{k!} \odd{f(a)}{x}{k}.
  \end{cases}
\end{align}
one may replace
linear difference operators $\delta, \Delta, \nabla$ with their non-linear dynamic
analogs, allowing more generic approach for sums of powers.
It is indeed so remarkable that Taylor's series
$f(x)= \sum_{k=0}^{\infty} \frac{(x-a)^k}{k!} \odd{f(a)}{x}{k}$
is a continuous analogue of Newton's formula!
Back to the topic, such non-linear difference operators arise naturally in the theory of dynamic equations on time scales,
developed by Bohner and Peterson in~\cite{Bohner2001DynamicEO}.
Thus, the discrete change $x+h$ of a function $f(x+h)$ is replaced by function $g(h)$ which
implies that dynamic difference operator is
\begin{align}
  D_g f = \frac{f(g(x)) - f(x)}{g(x) - x}.
  \label{eq:dynamic-difference-operator}
\end{align}
A well-known example is Jackson's $q$-difference~\cite{jackson_1909}
\begin{align*}
  D_q f(x) = \frac{f(qx) - f(x)}{qx - x},
\end{align*}
where $g(x) = qx$.
In particular, for Newton's formula in linear differences $\delta, \Delta, \nabla$, their respective
falling, rising, and central factorials are related
to binomial coefficients via
\begin{align}
  \label{eq:factorial-identities}
  \begin{cases}
    \frac{\fallingFactorial{x}{n}}{n!} &= \frac{1}{n!} x(x-1)(x-2)\cdots(x-n+1) =\binom{x}{n}, \\
    \frac{\risingFactorial{x}{n}}{n!}  &= \frac{1}{n!} x(x+1)(x+2)\cdots(x+n-1)  =\binom{x+n-1}{n}, \\
    \frac{\centralFactorial{x}{n}}{n!} &= \frac{1}{n!} \left( n + \frac{k}{2} -1 \right)\left( n + \frac{k}{2} -2 \right)
    \cdots \left( n - \frac{k}{2} +1 \right)  =\frac{x}{n} \binom{x+\frac{n}{2}-1}{n-1}.
  \end{cases}
\end{align}
Motivated by this viewpoint, we can see that dynamic version of Newton's formula
is
\begin{align}
  \label{eq:generalized-newtons-formula}
  f(x) = \sum_{k=0}^{\infty} \frac{\generalizedFactorial{x}{\alpha, \beta}{k}}{k!} \cdot D_g f(a),
\end{align}
where $a$ is an arbitrary integer, $D_g f(a)$ is dynamic difference operator,
and $\generalizedFactorial{x}{\alpha, \beta}{n} $
is generalized factorial which uses the combination of N\"orlund's notation~\cite{Norlund1924},
and definition given by Steffensen~\cite{steffensen1933definition}
\begin{definition}[Generalized factorial] For non-negative integers $x,n,\alpha,\beta$
  \label{prop:generalized-factorial}
  \begin{align*}
    \generalizedFactorial{x}{\alpha, \beta}{n} = x(x-n\alpha - \beta)(x-n\alpha - 2\beta)\cdots(x-n\alpha - n - I\beta).
  \end{align*}
\end{definition}
For instance,
\begin{align*}
  \begin{cases}
    \fallingFactorial{x}{n} &= \generalizedFactorial{x}{\alpha, \beta}{n}, \quad \alpha=0, \quad \beta =1, \quad \text{(forward Newton's formula)} \\
    \risingFactorial{x}{n}  &= \generalizedFactorial{x}{\alpha, \beta}{n}, \quad \alpha=0, \quad \beta =-1,  \quad \text{(backward Newton's formula)}  \\
    \centralFactorial{x}{n} &= \generalizedFactorial{x}{\alpha, \beta}{n}, \quad \alpha=\frac{1}{2}, \quad \beta =1,  \quad \text{(central Newton's formula)}\\
    x^n                     &= \generalizedFactorial{x}{\alpha, \beta}{n}, \quad \alpha=0, \quad \beta =0,  \quad \text{(Taylor's formula)}
  \end{cases}
\end{align*}
More precisely, the algorithm for closed formulas for sums of powers is
\begin{algorithm} [Algorithm for closed formulas]
  \label{alg:sums-of-powers}
  \leavevmode\par
  \begin{enumerate}
    \item Define the function $f(x) = x^m$.
    \item Pick generalized difference operator $D_g f(x)$, see~\eqref{eq:dynamic-difference-operator}.
    \item Derive generalized Newton's formula~\eqref{eq:generalized-newtons-formula}
      in terms of $D_g f(x)$, for example~\eqref{eq:newtons-formulas-casess}.
    \item Find an identity that connects respective factorials $\generalizedFactorial{x}{\alpha, \beta}{n}$
      with binomial coefficients of the form $\binom{x}{k}$,
      so that upper index is expression in $x$, for example~\eqref{eq:factorial-identities}.
    \item Apply hockey stick identities \eqref{lem:generalized-hockey-stick-identity},
      and
      \eqref{lem:centered-hockey-stick-identity} to derive closed form for sums of powers $\KnuthRFoldSum{1}{n}{m}$.
    \item Apply identities \eqref{lem:generalized-hockey-stick-identity},
      and
      \eqref{lem:centered-hockey-stick-identity} $r$-times to reach $r$-fold closed form for sums of powers $\KnuthRFoldSum{r}{n}{m}$.
  \end{enumerate}
\end{algorithm}
The algorithm above has already been utilized in practice.
For instance, the work~\cite{kolosov_2026_forward} provides remarkable
formulas for sums of powers, based on forward difference Newton's formula,
\begin{proposition}
  For integers $r\geq 0$, and $n\geq 0$, and $m\geq 0$,
  and an arbitrary integer $t$,
  \begin{align*}
    \KnuthRFoldSum{r}{n}{m}
    &= \sum_{j=0}^{m}
    \left[ \binom{n-t+r}{j+r} +
      \sum_{s=1}^{r} (-1)^{j+s-1} \binom{j+t-1}{j+s} \binom{r-s+n-1}{r-s}
    \right] \Delta^{j} t^{m}.
  \end{align*}
\end{proposition}
Also, in terms of Eulerian numbers $\eulerian{n}{k}$
\begin{proposition}
  For integers $r\geq0$, and $n\geq 0$, and $m\geq 0$,
  and an arbitrary integer $t$,
  \begin{align*}
    \KnuthRFoldSum{r}{n}{m}
    = \sum_{j=0}^{m} \sum_{k=0}^{m}
    \left[ \binom{n-t+r}{j+r} -
      \sum_{s=1}^{r} \binom{s-t}{j+s} \binom{r-s+n-1}{r-s}
    \right]
    \binom{t+k}{m-j} \eulerian{m}{k}.
  \end{align*}
\end{proposition}
The work entitled
\textit{Sums of powers of integers and generalized Stirling numbers of the second kind}
~\cite{cereceda2022sumspowersintegersgeneralized} discusses closed formulas for sums of powers
in terms of generalized Stirling numbers of the second kind,
which originates from forward Newton's formula for powers, around arbitrary
integer point $t$.
It is quite remarkable that both~\cite{cereceda2022sumspowersintegersgeneralized},
and~\cite{kolosov_2026_forward} reached the same result independently.
In~\cite{cereceda2022sumspowersintegersgeneralized}, we read
\begin{proposition}[Cereceda (2022)]
  \begin{align}
    \label{eq:cereceda-sums-1}
    \KnuthRFoldSum{1}{n}{k} = \sum_{j=0}^{k} j!
    \left[
      \binom{n+1-r}{j+1} + (-1)^j \binom{r+j-1}{j+1}
    \right] \stirlingii{k}{j}_{r},
  \end{align}
  \begin{align}
    \label{eq:cereceda-sums-2}
    \KnuthRFoldSum{1}{n}{k} = \sum_{j=0}^{k} j!
    \left[
      \binom{n+1+r}{j+1} + \binom{r+1}{j+1}
    \right] \stirlingii{k}{j}_{-r}.
  \end{align}
\end{proposition}
Formula~\eqref{eq:cereceda-sums-1}
is a special case of proposition~\cite[Prop. (1.9)]{kolosov_2026_forward}
\begin{proposition}
  For integers $n\geq 0$, and $m\geq 0$, and an arbitrary integer $t$
  \begin{align*}
    \KnuthRFoldSum{1}{n}{m} = \sum_{j=0}^{m}
    \left[ \binom{n-t+1}{j+1} + (-1)^j \binom{j+t-1}{j+1} \right] \Delta^{j} t^m.
  \end{align*}
\end{proposition}
in terms of generalized Stirling numbers $\stirlingii{k}{j}_{r}$.
This implies that finite differences of powers can be expressed
in terms of generalized Stirling numbers of the second kind,
$\Delta^{j} t^m = j! \stirlingii{m}{j}_{t}$.
Formula~\eqref{eq:cereceda-sums-2} is a special case
of proposition~\cite[Prop. (1.8)]{kolosov_2026_forward}
with setting $t=-r$
\begin{proposition}
  For integers $n\geq 0$, and $m\geq 0$, and an arbitrary integer $t$,
  \begin{align*}
    \KnuthRFoldSum{1}{n}{m} = \sum_{j=0}^{m}
    \left[ \binom{n+t+1}{j+1} - \binom{1+t}{j+1} \right] \Delta^{j} (-t)^m.
  \end{align*}
\end{proposition}
Moving discussion to backward Newton's formula, the work~\cite{kolosov_2026_backward}
utilizes algorithm~\eqref{alg:sums-of-powers} in terms of backward differences,
providing another remarkable formula
\begin{proposition}
  For non-negative integers $r,n,m$ and an arbitrary integer $t$,
  \begin{align*}
    \KnuthRFoldSum{r+1}{n}{m} = \sum_{j=0}^{m}
    \left[
      \binom{j-t+n+r}{j+r+1}
      + \sum_{s=0}^{r} (-1)^{j+s} \binom{t}{j+s+1} \binom{r-s+n-1}{r-s}
    \right] \nabla^{j} t^{m}.
  \end{align*}
\end{proposition}
Finally, the work~\cite{kolosov_2026_central} explores Newton's formula
in central differences, allowing it to be applied with algorithm~\eqref{alg:sums-of-powers}.
Essentially, it achieves similar formulas for sums of powers as current paper,
but using different binomial identities to reach the closed form.
For instance,
\begin{proposition}
  For integers $n\geq 0, \; m \geq 0$, and an arbitrary integer $t$
  \begin{align*}
    \KnuthRFoldSum{r}{n}{m}
    &= \binom{r+n-1}{r} t^m
    + \sum_{k=1}^{m} \frac{\delta^{k} t^{m}}{2}
    \Bigg \{
      \left[
        \binom{n-t+\frac{k}{2}+r}{k+r}
        +
        \binom{n-t+\frac{k}{2}+r-1}{k+r}
      \right] \\
      &-
      \sum_{s=0}^{r-1}
      \left[
        \binom{-t+\frac{k}{2} + r -s}{k+r-s} +
        \binom{-t+\frac{k}{2} + r -s-1}{k+r-s}
      \right] \binom{s+n-1}{s}
    \Bigg \}.
  \end{align*}
\end{proposition}
