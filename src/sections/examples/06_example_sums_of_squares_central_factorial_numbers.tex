\begin{example} [Sums of squares in central factorial numbers]
  \begin{align*}
    t=0:\quad
    \KnuthRFoldSum{1}{n}{2}
    &= n\cdot0 + \tfrac12 n(1+n)\cdot0 + \tfrac16 n(1+3n+2n^2)\cdot1,
    \\[4pt]
    %
    t=1:\quad
    \KnuthRFoldSum{1}{n}{2}
    &= n\cdot1 + \tfrac12 n(n-1)\cdot2 + \tfrac16 n(1-3n+2n^2)\cdot1,
    \\[4pt]
    %
    t=2:\quad
    \KnuthRFoldSum{1}{n}{2}
    &= n\cdot4 + \tfrac12 n(n-3)\cdot4 + \tfrac16 n(13-9n+2n^2)\cdot1,
    \\[4pt]
    %
    t=3:\quad
    \KnuthRFoldSum{1}{n}{2}
    &= n\cdot9 + \tfrac12 n(n-5)\cdot6 + \tfrac16 n(37-15n+2n^2)\cdot1,
    \\[4pt]
    %
    t=4:\quad
    \KnuthRFoldSum{1}{n}{2}
    &= n\cdot16 + \tfrac12 n(n-7)\cdot8 + \tfrac16 n(73-21n+2n^2)\cdot1,
    \\[4pt]
    %
    t=5:\quad
    \KnuthRFoldSum{1}{n}{2}
    &= n\cdot25 + \tfrac12 n(n-9)\cdot10 + \tfrac16 n(121-27n+2n^2)\cdot1.
  \end{align*}
\end{example}
