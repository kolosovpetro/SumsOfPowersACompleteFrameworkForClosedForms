\begin{example}[Double sum of squares in central differences]
  \begin{align*}
    t=0:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \frac{t^2}{2}\, n(1+n)
    +
    \frac{\delta t^2}{2}\,\frac{n(2+3n+n^2)}{3}
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(1+n)^2(2+n)}{12},
    \\
    %
    t=1:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \frac{t^2}{2}\, n(1+n)
    +
    \frac{\delta t^2}{2}\,\frac{n(-1+n^2)}{3}
    +
    \frac{\delta^2 t^2}{2}\,\frac{n^2(-1+n^2)}{12},
    \\
    %
    t=2:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \frac{t^2}{2}\, n(1+n)
    +
    \frac{\delta t^2}{2}\,\frac{n(-4-3n+n^2)}{3}
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(10+5n-4n^2+n^3)}{12},
    \\
    %
    t=3:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \frac{t^2}{2}\, n(1+n)
    +
    \frac{\delta t^2}{2}\,\frac{n(-7-6n+n^2)}{3}
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(32+23n-8n^2+n^3)}{12},
    \\
    %
    t=4:\quad
    \KnuthRFoldSum{2}{n}{2}
    &=
    \frac{t^2}{2}\, n(1+n)
    +
    \frac{\delta t^2}{2}\,\frac{n(-10-9n+n^2)}{3}
    +
    \frac{\delta^2 t^2}{2}\,\frac{n(66+53n-12n^2+n^3)}{12}.
  \end{align*}
\end{example}
