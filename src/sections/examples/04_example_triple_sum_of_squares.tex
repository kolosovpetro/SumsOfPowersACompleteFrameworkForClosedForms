\begin{example}[Triple sum of squares in central differences]
  For integer $0 \leq t \leq 4$
  \begin{align*}
    t=0:\quad
    \KnuthRFoldSum{3}{n}{2}
    &=
    \tfrac{t^2}{2}\,\tfrac{n(2+3n+n^2)}{3}
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(6+11n+6n^2+n^3)}{12}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(18+45n+40n^2+15n^3+2n^4)}{120},
    \\
    %
    t=1:\quad
    \KnuthRFoldSum{3}{n}{2}
    &=
    \tfrac{t^2}{2}\,\tfrac{n(2+3n+n^2)}{3}
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-2-n+2n^2+n^3)}{12}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(-2-5n+5n^3+2n^4)}{120},
    \\
    %
    t=2:\quad
    \KnuthRFoldSum{3}{n}{2}
    &=
    \tfrac{t^2}{2}\,\tfrac{n(2+3n+n^2)}{3}
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-10-13n-2n^2+n^3)}{12}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(58+65n-5n^3+2n^4)}{120},
    \\
    %
    t=3:\quad
    \KnuthRFoldSum{3}{n}{2}
    &=
    \tfrac{t^2}{2}\,\tfrac{n(2+3n+n^2)}{3}
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-18-25n-6n^2+n^3)}{12}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(198+255n+40n^2-15n^3+2n^4)}{120},
    \\
    %
    t=4:\quad
    \KnuthRFoldSum{3}{n}{2}
    &=
    \tfrac{t^2}{2}\,\tfrac{n(2+3n+n^2)}{3}
    +
    \tfrac{\delta t^2}{2}\,\tfrac{n(-26-37n-10n^2+n^3)}{12}
    +
    \tfrac{\delta^2 t^2}{2}\,\tfrac{n(418+565n+120n^2-25n^3+2n^4)}{120}.
  \end{align*}
\end{example}
