By decomposition of Newton's formula
for powers~\eqref{lem:central-newtons-formula-for-powers-decomposition},
for integers $n,m \geq 0$, and an arbitrary integer $t$, we get
\begin{align}
  \label{eq:ordinary-sums-of-powers}
  \KnuthRFoldSum{1}{n}{m} = \frac{1}{2} \sum_{k=0}^{m}
  \sum_{j=1}^{n}
  \left[
    \binom{j-t+\frac{k}{2}}{k} + \binom{j-t+\frac{k}{2}-1}{k}
  \right] \delta^{k} t^m.
\end{align}
Now we reach the transition to closed form of ordinary sums of powers,
by setting $r=-1$, and $r=-2$ into
Lemma~\eqref{lem:centered-hockey-stick-identity}.
Therefore,
\begin{proposition}[Ordinary sums of powers decomposition]
  \label{prop:ordinary-sums-of-powers-decomposition}
  For integers $n,m \geq 0$, and an arbitrary integer $t$,
  \begin{align*}
    \KnuthRFoldSum{1}{n}{m} = \frac{1}{2} \sum_{k=0}^{m}
    \left[
      \binom{n -t + \frac{k}{2} +1}{k+1} - \binom{1 -t + \tfrac{k}{2}}{k+1}
      + \binom{n -t + \frac{k}{2}}{k+1} - \binom{-t + \tfrac{k}{2}}{k+1}
    \right] \delta^{k} t^m.
  \end{align*}
  \begin{proof}
    We have the sums of powers expressed by~\eqref{eq:ordinary-sums-of-powers},
    thus,
    \begin{align*}
      \tsum_{j=1}^{n}
      \left[
        \tbinom{j-t+\tfrac{k}{2}}{k} + \tbinom{j-t+\tfrac{k}{2}-1}{k}
      \right] = \tbinom{n -t + \tfrac{k}{2} +1}{k+1} - \tbinom{1 -t + \tfrac{k}{2}}{k+1}
      + \tbinom{n -t + \tfrac{k}{2}}{k+1} - \tbinom{-t + \tfrac{k}{2}}{k+1},
    \end{align*}
    by setting $r=-1$, and $r=-2$ into Lemma~\eqref{lem:centered-hockey-stick-identity}.
  \end{proof}
\end{proposition}

It is indeed interesting to notice the connection between
the decomposition of ordinary sums of
powers~\eqref{prop:ordinary-sums-of-powers-decomposition}
and Riordan's central factorial numbers of the second kind, that is
\begin{proposition}[Central factorial numbers sums]
  \label{prop:central-factorial-numbers-sums}
  For integers $n,m \geq 0$, and an arbitrary integer $t$,
  \begin{align*}
    \KnuthRFoldSum{1}{n}{m} = \frac{1}{2} \sum_{k=0}^{m}
    \left[
      \binom{n + \frac{k}{2} +1}{k+1} - \binom{1 + \tfrac{k}{2}}{k+1}
      + \binom{n + \frac{k}{2}}{k+1} - \binom{\tfrac{k}{2}}{k+1}
    \right] k! T(m,k),
  \end{align*}
  where $T(n,k)$ denotes the central factorial numbers
  of the second kind
  (in Riordan's notation). See~\cite[p. 213]{riordan1968combinatorial},
  and~\cite{carlitz1963divided}.
\end{proposition}
These central factorial numbers of the second kind $T(n,k)$ are
defined by central Newton's formula for
powers~\eqref{lem:central-newtons-formula-for-powers}
evaluated in zero, such that
\begin{align*}
  n^m = \sum_{k=0}^{\infty} T(m,k) \centralFactorial{n}{k}.
\end{align*}
Which implies
\begin{align*}
  T(n,k) = \frac{\delta^{k} 0^m}{k!},
\end{align*}
by Newton's
formula~\eqref{lem:central-newtons-formula-for-powers}.
