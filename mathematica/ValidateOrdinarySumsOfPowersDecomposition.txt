(* ================================================================ *)
(*  Ordinary Sums of Powers Decomposition Validation                *)
(* ================================================================ *)

ClearAll[
  MultifoldSumOfPowersRecurrence,
  CentralDifference,
  OrdinarySumsOfPowersDecomposition,
  ValidateOrdinarySumsOfPowersDecomposition
];

(* --------------------------------------------------------------- *)
(*  Convention: 0^x = 1                                             *)
(* --------------------------------------------------------------- *)

Begin["`Private`"]

Unprotect[Power];
Power[0 | 0., 0 | 0.] = 1;
Protect[Power];

End[];


(* --------------------------------------------------------------- *)
(*  r-fold sums of powers                                           *)
(* --------------------------------------------------------------- *)

MultifoldSumOfPowersRecurrence[r_, n_, m_] := 0;

MultifoldSumOfPowersRecurrence[0, n_, m_] :=
  n^m;

MultifoldSumOfPowersRecurrence[r_, n_, m_] :=
  Sum[
    MultifoldSumOfPowersRecurrence[r - 1, k, m],
    {k, 1, n}
  ] /; r > 0;


(* --------------------------------------------------------------- *)
(*  Central difference operator                                     *)
(* --------------------------------------------------------------- *)

CentralDifference[x_, n_, k_] :=
  Sum[
    (-1)^j
      Binomial[k, j]
      (x + k/2 - j)^n,
    {j, 0, k}
  ];


(* --------------------------------------------------------------- *)
(*  Ordinary sums of powers decomposition                           *)
(* --------------------------------------------------------------- *)

OrdinarySumsOfPowersDecomposition[n_, m_, t_] :=
  1/2 *
    Sum[
      (
        Binomial[n - t + k/2 + 1, k + 1]
        - Binomial[1 - t + k/2, k + 1]
        + Binomial[n - t + k/2, k + 1]
        - Binomial[-t + k/2, k + 1]
      )
      *
      CentralDifference[t, m, k],
      {k, 0, m}
    ];


(* --------------------------------------------------------------- *)
(*  Validation routine                                              *)
(* --------------------------------------------------------------- *)

ValidateOrdinarySumsOfPowersDecomposition[max_] :=
  Flatten[
    Table[
      FullSimplify[
        MultifoldSumOfPowersRecurrence[1, n, m]
          -
        OrdinarySumsOfPowersDecomposition[n, m, t]
      ],
      {n, 0, max},
      {m, 0, max},
      {t, -max, max}
    ]
  ];

ValidateOrdinarySumsOfPowersDecomposition[10]
